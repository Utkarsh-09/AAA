import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { Button, Input, Typography, Container } from '@mui/material';
import CloudUploadIcon from '@mui/icons-material/CloudUpload';
import './Upload.css';

const Upload = () => {
  const [file, setFile] = useState(null);
  const navigate = useNavigate();

  const handleFileChange = (e) => {
    setFile(e.target.files[0]);
  };

  const handleUpload = () => {
    setTimeout(() => {
      console.log("File uploaded:", file);
      navigate('/app');
    }, 1000);
  };

  return (
    <Container maxWidth="sm" className="root">
      <Typography variant="h4" gutterBottom className="upload-title">
        Upload Audio File
      </Typography>
      <Input
        accept="audio/*"
        id="contained-button-file"
        type="file"
        style={{ display: 'none' }}
        onChange={handleFileChange}
      />
      <label htmlFor="contained-button-file" className="upload-btn-wrapper">
        <Button
          variant="contained"
          component="span"
          startIcon={<CloudUploadIcon />}
          size="large"
        >
          Choose File
        </Button>
      </label>
      <Button
        variant="outlined"
        onClick={handleUpload}
        disabled={!file}
        className="upload-file-btn"
        size="large"
      >
        Upload
      </Button>
    </Container>
  );
};

export default Upload;

</Router>,
